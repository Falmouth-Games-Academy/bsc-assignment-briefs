	\section{This algorithm copies the top of an image to the bottom}
	\begin{algorithm}[ht]
	\caption{Top-Copy}
	\label{alg:algorithm}
	
	\begin{algorithmic}[1]
		   \State $ht \leftarrow \frac{h}{2}$
			\For {$y = 0, y < ht$, $x=0, x < w$}
			\State $i \leftarrow ht + y$
			\State pixel(x, i) $\leftarrow$ pixel(x, y)
			\EndFor
	\end{algorithmic}
	
\end{algorithm}
